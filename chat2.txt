    <script>
        const selectedMessages = new Set();
        let userScrolledUp = false;

        function toggleMessageSelection(messageElement, messageId) {
            if (selectedMessages.has(messageId)) {
                selectedMessages.delete(messageId);
                messageElement.classList.remove('selected');
            } else {
                selectedMessages.add(messageId);
                messageElement.classList.add('selected');
            }
            updateDeleteButtonVisibility();
        }

        function updateDeleteButtonVisibility() {
            const deleteButtonContainer = document.getElementById('deleteSelectedContainer');
            if (selectedMessages.size > 0) {
                deleteButtonContainer.style.display = 'block';
            } else {
                deleteButtonContainer.style.display = 'none';
            }
        }

        function fetchMessages() {
            fetch('{{ url_for("get_messages", recipient=recipient) }}')
                .then(response => response.json())
                .then(data => {
                    console.log('Fetched messages:', data);
                    const messagesList = document.getElementById('messages-list');
                    const existingMessageIds = new Set(Array.from(messagesList.children).map(li => li.getAttribute('data-id')));

                    data.forEach(message => {
                        if (!existingMessageIds.has(message.id.toString())) {
                            const li = document.createElement('li');
                            li.className = 'list-group-item';
                            li.id = `message-${message.id}`;
                            li.classList.add(message.sender === "{{ session['username'] }}" ? 'text-right' : 'text-left');
                            li.style.borderRadius = '15px';
                            li.style.marginBottom = '10px';
                            li.setAttribute('data-id', message.id);
                            li.setAttribute('ondblclick', `toggleMessageSelection(this, '${message.id}')`);
                            li.innerHTML = `${message.content}`;
                            if (message.image_url) {
                                li.innerHTML += `<br><img src="${message.image_url}" alt="Image" class="img-thumbnail" style="max-width: 150px; border-radius: 15px;" onclick="openImageModal('${message.image_url}')">`;
                            }
                            if (message.file_url) {
                                if (message.file_url.endsWith('.mp4') || message.file_url.endsWith('.webm')) {
                                    li.innerHTML += `<br><video controls style="max-width: 150px; border-radius: 15px;"><source src="${message.file_url}" type="video/mp4"></video>`;
                                } else if (message.file_url.endsWith('.mp3') || message.file_url.endsWith('.wav')) {
                                    li.innerHTML += `<br><audio controls style="max-width: 150px; border-radius: 15px;"><source src="${message.file_url}" type="audio/mpeg"></audio>`;
                                } else {
                                    li.innerHTML += `<br><a href="${message.file_url}" target="_blank">${message.file_name}</a>`;
                                }
                            }
                            li.innerHTML += `<br><small class="text-muted">${message.timestamp}</small>`;
                            messagesList.appendChild(li);
                        }
                    });

                    if (!userScrolledUp) {
                        scrollToBottom();
                    }
                })
                .catch(error => {
                    console.error('Error fetching messages:', error);
                });
        }

        document.addEventListener('DOMContentLoaded', function() {
            fetchMessages();
            setInterval(fetchMessages, 3000);
        });

        function deleteSelectedMessages() {
            const messageIds = Array.from(selectedMessages);

            fetch('/delete_messages', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({ message_ids: messageIds })
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    messageIds.forEach(id => {
                        const messageElement = document.getElementById(`message-${id}`);
                        messageElement.remove();
                    });
                    selectedMessages.clear();
                    updateDeleteButtonVisibility();
                }
            })
            .catch(error => {
                console.error('Error deleting messages:', error);
            });
        }

        function previewFile(event) {
            const filePreview = document.getElementById('filePreview');
            filePreview.innerHTML = '';
            const file = event.target.files[0];
            if (file && file.size <= 20 * 1024 * 1024) { // Check file size is under 20 MB
                const fileName = document.createElement('p');
                fileName.textContent = file.name;
                filePreview.appendChild(fileName);
            } else {
                alert('File size must be under 20 MB');
            }
        }

        function deleteMessage(messageId) {
            fetch(`/delete_message/${messageId}`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                }
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    const messageElement = document.getElementById(`message-${messageId}`);
                    messageElement.remove();
                }
            });
        }

        function banUser(userId) {
            fetch(`/ban_user/${userId}`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                }
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    const userElement = document.querySelector(`li a[href$="/chat/${userId}"]`).parentElement;
                    userElement.remove();
                }
            });
        }

        function searchUsers() {
            const searchInput = document.getElementById('user-search').value.toLowerCase();
            const userList = document.getElementById('user-list');
            const users = userList.getElementsByTagName('li');

            for (let i = 0; i < users.length; i++) {
                const user = users[i].getElementsByTagName('a')[0];
                const username = user.textContent || user.innerText;
                if (username.toLowerCase().indexOf(searchInput) > -1) {
                    users[i].style.display = '';
                } else {
                    users[i].style.display = 'none';
                }
            }
        }

        function openImageModal(imageUrl) {
            const modalImage = document.getElementById('modalImage');
            modalImage.src = imageUrl;
            $('#imageModal').modal('show');
        }

        function scrollToBottom() {
            const chatBody = document.getElementById('chat-body');
            chatBody.scrollTop = chatBody.scrollHeight;
        }

        document.addEventListener('DOMContentLoaded', function() {
            scrollToBottom();

            const chatBody = document.getElementById('chat-body');
            chatBody.addEventListener('scroll', function() {
                const isScrolledToBottom = chatBody.scrollHeight - chatBody.clientHeight <= chatBody.scrollTop + 1;
                userScrolledUp = !isScrolledToBottom;
            });

            const form = document.getElementById('messageForm');
            form.addEventListener('submit', function(event) {
                event.preventDefault();

                const formData = new FormData(form);
                const uploadProgress = document.getElementById('uploadProgress');
                const progressBar = uploadProgress.querySelector('.progress-bar');

                const xhr = new XMLHttpRequest();
                xhr.open('POST', form.action, true);

                xhr.upload.addEventListener('progress', function(event) {
                    if (event.lengthComputable) {
                        const percentComplete = (event.loaded / event.total) * 100;
                        progressBar.style.width = percentComplete + '%';
                        progressBar.setAttribute('aria-valuenow', percentComplete);
                    }
                });

                xhr.addEventListener('load', function() {
                    if (xhr.status === 200) {
                        const response = JSON.parse(xhr.responseText);
                        if (response && response.success && response.message) {
                            const message = response.message;
                            const messagesList = document.getElementById('messages-list');
                            const li = document.createElement('li');
                            li.className = 'list-group-item ' + (message.sender === "{{ session['username'] }}" ? 'text-right' : 'text-left');
                            li.id = `message-${message.id}`;
                            li.setAttribute('data-id', message.id);
                            li.style.borderRadius = '15px';
                            li.style.marginBottom = '10px';
                            li.ondblclick = function() { toggleMessageSelection(this, message.id); };
                            li.innerHTML = `${message.content}`;
                            if (message.image_url) {
                                li.innerHTML += `<br><img src="${message.image_url}" alt="Image" class="img-thumbnail" style="max-width: 150px; border-radius: 15px;" onclick="openImageModal('${message.image_url}')">`;
                            }
                            if (message.file_url) {
                                if (message.file_url.endsWith('.mp4') || message.file_url.endsWith('.webm')) {
                                    li.innerHTML += `<br><video controls style="max-width: 150px; border-radius: 15px;"><source src="${message.file_url}" type="video/mp4"></video>`;
                                } else if (message.file_url.endsWith('.mp3') || message.file_url.endsWith('.wav')) {
                                    li.innerHTML += `<br><audio controls style="max-width: 150px; border-radius: 15px;"><source src="${message.file_url}" type="audio/mpeg"></audio>`;
                                } else {
                                    li.innerHTML += `<br><a href="${message.file_url}" target="_blank">${message.file_name}</a>`;
                                }
                            }
                            li.innerHTML += `<br><small class="text-muted">${message.timestamp}</small>`;
                            messagesList.appendChild(li);

                            document.querySelector('input[name="message"]').value = '';
                            document.querySelector('input[name="file"]').value = '';
                            document.getElementById('filePreview').innerHTML = '';

                            scrollToBottom();
                        }
                    } else {
                        console.error('Error sending message:', xhr.statusText);
                    }
                    uploadProgress.style.display = 'none';
                    progressBar.style.width = '0%';
                    progressBar.setAttribute('aria-valuenow', '0');
                });

                xhr.addEventListener('error', function() {
                    console.error('Error sending message:', xhr.statusText);
                    uploadProgress.style.display = 'none';
                    progressBar.style.width = '0%';
                    progressBar.setAttribute('aria-valuenow', '0');
                });

                uploadProgress.style.display = 'block';
                xhr.send(formData);
            });
        });

        fetchMessages();
        
        function openManageUserModal() {
            $('#manageUserModal').modal('show');
        }
    </script>
{% endblock %}
