    <script>
    const selectedMessages = new Set();
    let userScrolledUp = false;
    let replyMessageId = null;
    let formSubmitted = false;

    function toggleMessageSelection(messageElement, messageId) {
        if (selectedMessages.has(messageId)) {
            selectedMessages.delete(messageId);
            messageElement.classList.remove('selected');
        } else {
            selectedMessages.add(messageId);
            messageElement.classList.add('selected');
        }
        updateDeleteButtonVisibility();
    }

    function updateDeleteButtonVisibility() {
        const deleteButtonContainer = document.getElementById('deleteSelectedContainer');
        if (selectedMessages.size > 0) {
            deleteButtonContainer.style.display = 'block';
        } else {
            deleteButtonContainer.style.display = 'none';
        }
    }

    function fetchMessages() {
        fetch('{{ url_for("get_messages", recipient=recipient) }}')
            .then(response => response.json())
            .then(data => {
                const messagesList = document.getElementById('messages-list');
                const existingMessageIds = new Set(Array.from(messagesList.children).map(li => li.getAttribute('data-id')));

                data.forEach(message => {
                    const messageId = message.id.toString();
                    if (!document.getElementById(`message-${messageId}`) && !existingMessageIds.has(messageId)) {
                        const li = document.createElement('li');

    function desktopSearchUsers() {
        const searchInput = document.getElementById('desktop-user-search').value.toLowerCase();
        const userList = document.getElementById('desktop-user-list');
        const users = userList.getElementsByTagName('li');
        const searchButton = document.querySelector('.search-btn i.fas.fa-search');
        const searchStatus = document.getElementById('search-status');
        const searchTermDisplay = document.getElementById('search-term-display');
        const noResults = document.getElementById('no-results-message');
        
        // Add loading state
        searchButton.classList.add('fa-spin');

        // First hide all users
        Array.from(users).forEach(user => {
            user.style.display = 'none';
            user.classList.remove('search-match');
        });

        if (searchInput.trim() === '') {
            // Show all users if search is empty
            Array.from(users).forEach(user => {
                user.style.display = '';
            });
            searchStatus.style.display = 'none';
            noResults.style.display = 'none';
        } else {
            let exactMatches = [];
            let partialMatches = [];
            
            Array.from(users).forEach(user => {
                const username = user.querySelector('.user-name').textContent.toLowerCase();
                if (username === searchInput) {
                    exactMatches.push(user);
                } else if (username.includes(searchInput)) {
                    partialMatches.push(user);
                }
            });

            // Show matches in order (exact first, then partial)
            [...exactMatches, ...partialMatches].forEach(user => {
                user.style.display = '';
                user.classList.add('search-match');
                // Move matched users to top
                userList.insertBefore(user, userList.firstChild);
            });

            // Update search status
            if (exactMatches.length > 0 || partialMatches.length > 0) {
                searchStatus.style.display = 'block';
                searchTermDisplay.textContent = searchInput;
                noResults.style.display = 'none';
            } else {
                searchStatus.style.display = 'none';
                noResults.style.display = 'block';
            }
        }

        // Remove loading state
        setTimeout(() => {
            searchButton.classList.remove('fa-spin');
        }, 300);
    }
